# 快速上手

# 添加 CSS样式

*,*::before,*::after{box-sizing: border-box;}

IE8 及 以上浏览器都支持此样式

# 引入 gulu

import {Button} from 'gulu'
import Vue from 'vue'

new Vue({
  el: '#app',
  components: {
    'g-button':Button
  }
})
